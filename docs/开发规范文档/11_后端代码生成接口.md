# 11. 后端代码生成接口

## 11.1 后端代码生成概述

后端代码生成器负责将 MLIR 或 LLVM IR 转换为特定目标平台的机器码、汇编代码或字节码。幻语编译器采用模块化后端设计，支持多种目标平台：

- **LLVM 后端**：生成 x86、ARM、RISC-V 等架构的原生机器码
- **WebAssembly 后端**：生成 Wasm 字节码，用于浏览器和边缘计算
- **GPU 后端**：生成 CUDA、HIP、OpenCL 等 GPU 内核代码

所有后端均实现统一的 `CodeGenerator` 接口，通过命令行参数 `--目标` 选择。

### 11.2.1 CodeGenerator Trait

```rust
/// 代码生成器统一接口
pub trait CodeGenerator {
    /// 初始化代码生成器
    fn new(target: TargetTriple, options: CodeGenOptions) -> Self;

    /// 从 MLIR 模块生成目标代码
    fn generate(&mut self, mlir_module: &mlir::ModuleOp) -> Result<Vec<u8>, CodeGenError>;

    /// 生成汇编代码

    /// 生成 LLVM IR 文本（仅 LLVM 后端支持）

        Err(CodeGenError::Unsupported("LLVM IR 文本生成不支持此目标".to_string()))
    }

    /// 链接目标文件
    fn link(&self, objects: Vec<PathBuf>, output: PathBuf) -> Result<(), LinkError>;
}

/// 代码生成选项
#[derive(Debug, Clone)]
pub struct CodeGenOptions {
    /// 优化级别：0-3, 's' (尺寸), 'z' (激进尺寸)
    pub opt_level: OptLevel,
    /// 是否生成调试信息
    pub debug_info: bool,
    /// 是否生成位置无关代码（PIC/PIE）
    pub pic: bool,
    /// 是否启用栈保护
    pub stack_protector: bool,
    /// 目标 CPU 特性（如 "+avx2,+fma"）
    pub target_features: String,
    /// 调用约定
    pub calling_convention: CallingConvention,
    /// 是否保留内联汇编
    pub keep_asm: bool,
}

#[derive(Debug, Clone, Copy)]
pub enum OptLevel {
    None = 0,
    Less = 1,
    Default = 2,
    Aggressive = 3,
    Size = b's' as isize,
    SizeAggressive = b'z' as isize,
}

#[derive(Debug, Clone)]
pub enum CallingConvention {
    HuanDefault,
    C,
    Fast,
    Interrupt,
}

/// 目标三元组
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct TargetTriple(String);

impl TargetTriple {
    pub fn host() -> Self {
        Self(std::env::consts::ARCH.to_string() + "-unknown-" + std::env::consts::OS)
    }

    pub fn as_str(&self) -> &str {
        &self.0
    }

    pub fn arch(&self) -> Arch {
        // 解析三元组返回架构
        if self.0.starts_with("x86_64") { Arch::X86_64 }
        else if self.0.starts_with("aarch64") { Arch::AArch64 }
        else if self.0.starts_with("arm") || self.0.starts_with("thumb") { Arch::ARM }
        else if self.0.starts_with("riscv32") { Arch::RiscV32 }
        else if self.0.starts_with("riscv64") { Arch::RiscV64 }
        else if self.0.starts_with("wasm32") { Arch::Wasm32 }
        else { Arch::Unknown }
    }
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum Arch {
    X86_64, AArch64, ARM, RiscV32, RiscV64, Wasm32, Unknown,
}
```

## 11.3 LLVM 后端

LLVM 后端是幻语的主要原生代码生成后端，支持 x86、ARM、RISC-V 等主流 CPU 架构。

### 11.3.1 LLVM 后端实现结构

```rust
use inkwell::context::Context;
use inkwell::module::Module;
use inkwell::builder::Builder;
use inkwell::execution_engine::ExecutionEngine;
use inkwell::targets::{Target, TargetMachine, FileType, RelocMode, CodeModel};

pub struct LLVMBackend<'ctx> {
    /// LLVM 上下文
    context: &'ctx Context,
    /// LLVM 模块
    module: Module<'ctx>,
    /// 构建器
    builder: Builder<'ctx>,
    /// 目标机器
    target_machine: TargetMachine,
    /// 代码生成选项
    options: CodeGenOptions,
    /// 当前函数
    current_function: Option<FunctionValue<'ctx>>,
    /// 变量映射（名称 → 栈分配地址）
    variables: HashMap<String, PointerValue<'ctx>>,
    /// 类型缓存
    type_cache: HashMap<Type, BasicTypeEnum<'ctx>>,
}

impl<'ctx> CodeGenerator for LLVMBackend<'ctx> {
    fn new(target: TargetTriple, options: CodeGenOptions) -> Self {
        let context = Box::leak(Box::new(Context::create()));
        let module = context.create_module("huan_module");
        let builder = context.create_builder();

        // 初始化 LLVM 目标
        Target::initialize_native(&inkwell::targets::InitializationConfig::default())
            .expect("Failed to initialize native target");

        let target_machine = Target::from_triple(&target.0)
            .expect("Failed to get target")
            .create_target_machine(
                &target.0,
                "generic",
                &options.target_features,
                to_llvm_opt_level(options.opt_level),
                RelocMode::Default,
                CodeModel::Default,
            )
            .expect("Failed to create target machine");

        Self {
            context,
            module,
            builder,
            target_machine,
            options,
            current_function: None,
            variables: HashMap::new(),
            type_cache: HashMap::new(),
        }
    }

    fn generate(&mut self, mlir_module: &mlir::ModuleOp) -> Result<Vec<u8>, CodeGenError> {
        // 1. 将 MLIR 降级到 LLVM IR（此步骤由 MLIR Pass 管线完成）
        // 2. 生成目标代码
        let memory_buffer = self.target_machine
            .write_to_memory_buffer(&self.module, FileType::Object)
            .map_err(|e| CodeGenError::LlvmError(e.to_string()))?;
        
        Ok(memory_buffer.as_slice().to_vec())
    }

    fn emit_assembly(&mut self, mlir_module: &mlir::ModuleOp) -> Result<String, CodeGenError> {
        let memory_buffer = self.target_machine
            .write_to_memory_buffer(&self.module, FileType::Assembly)
            .map_err(|e| CodeGenError::LlvmError(e.to_string()))?;
        
        Ok(String::from_utf8_lossy(memory_buffer.as_slice()).to_string())
    }

    fn emit_llvm_ir(&mut self, _mlir_module: &mlir::ModuleOp) -> Result<String, CodeGenError> {
        Ok(self.module.print_to_string().to_string())
    }

    fn link(&self, objects: Vec<PathBuf>, output: PathBuf) -> Result<(), LinkError> {
        // 调用系统链接器（ld, lld, link.exe）
        link_objects(&objects, &output, &self.target_machine.get_triple())
    }
}
```

### 11.3.2 类型转换

```rust
impl<'ctx> LLVMBackend<'ctx> {
    /// 将幻语类型转换为 LLVM 类型
    fn convert_type(&mut self, ty: &Type) -> Result<BasicTypeEnum<'ctx>, CodeGenError> {
        if let Some(cached) = self.type_cache.get(ty) {
            return Ok(*cached);
        }

        let llvm_ty = match ty {
            Type::Int => self.context.i64_type().into(),
            Type::I8 => self.context.i8_type().into(),
            Type::I16 => self.context.i16_type().into(),
            Type::I32 => self.context.i32_type().into(),
            Type::I64 => self.context.i64_type().into(),
            Type::U8 => self.context.i8_type().into(),
            Type::U16 => self.context.i16_type().into(),
            Type::U32 => self.context.i32_type().into(),
            Type::U64 => self.context.i64_type().into(),
            Type::F32 => self.context.f32_type().into(),
            Type::F64 => self.context.f64_type().into(),
            Type::Bool => self.context.bool_type().into(),
            Type::Char => self.context.i32_type().into(), // Unicode 码点
            Type::String => {
                // 字符串：{ i8*, i64 }
                let ptr = self.context.i8_type().ptr_type(AddressSpace::default());
                let len = self.context.i64_type();
                self.context.struct_type(&[ptr.into(), len.into()], false).into()
            }
            Type::Unit => self.context.void_type().into(),
            Type::List(inner) => {
                // 列表：{ i8*, i64, i64, i64 }
                let ptr = self.context.i8_type().ptr_type(AddressSpace::default());
                let int = self.context.i64_type();
                self.context.struct_type(&[ptr.into(), int.into(), int.into(), int.into()], false).into()
            }
            Type::Ptr(inner) => {
                let inner_ty = self.convert_type(inner)?;
                inner_ty.ptr_type(AddressSpace::default()).into()
            }
            Type::Func(params, ret) => {
                let param_tys: Vec<BasicTypeEnum> = params
                    .iter()
                    .map(|p| self.convert_type(p))
                    .collect::<Result<_, _>>()?;
                let ret_ty = self.convert_type(ret)?;
                ret_ty.fn_type(&param_tys, false).into()
            }
            // ... 其他类型
            _ => return Err(CodeGenError::Unsupported(format!("类型 {:?} 不支持", ty))),
        };

        self.type_cache.insert(ty.clone(), llvm_ty);
        Ok(llvm_ty)
    }
}
```

### 11.3.3 函数编译

```rust
impl<'ctx> LLVMBackend<'ctx> {
    /// 编译函数定义
    fn compile_function(&mut self, func: &Function) -> Result<FunctionValue<'ctx>, CodeGenError> {
        let return_ty = self.convert_type(&func.return_type)?;
        let param_tys: Vec<BasicTypeEnum> = func.params
            .iter()
            .map(|(_, ty)| self.convert_type(ty))
            .collect::<Result<_, _>>()?;

        let fn_type = match return_ty {
            BasicTypeEnum::IntType(t) => t.fn_type(&param_tys, false),
            BasicTypeEnum::FloatType(t) => t.fn_type(&param_tys, false),
            BasicTypeEnum::PointerType(t) => t.fn_type(&param_tys, false),
            BasicTypeEnum::StructType(t) => t.fn_type(&param_tys, false),
            _ => self.context.void_type().fn_type(&param_tys, false),
        };

        let function = self.module.add_function(&func.name.name, fn_type, None);
        self.current_function = Some(function);

        // 创建入口基本块
        let entry = self.context.append_basic_block(function, "entry");
        self.builder.position_at_end(entry);

        // 清空局部变量
        self.variables.clear();

        // 分配参数
        for (i, (name, ty)) in func.params.iter().enumerate() {
            let param = function.get_nth_param(i as u32).unwrap();
            let alloca = self.builder.build_alloca(param.get_type(), &name.name)?;
            self.builder.build_store(alloca, param)?;
            self.variables.insert(name.name.clone(), alloca);
        }

        // 编译函数体
        for stmt in &func.body {
            self.compile_stmt(stmt)?;
        }

        // 确保函数有返回
        self.ensure_return(&func.return_type)?;

        self.current_function = None;
        Ok(function)
    }

    fn compile_stmt(&mut self, stmt: &Stmt) -> Result<(), CodeGenError> {
        match stmt {
            Stmt::Let { name, value, .. } => {
                let val = self.compile_expr(value)?;
                let alloca = self.builder.build_alloca(val.get_type(), &name.name)?;
                self.builder.build_store(alloca, val)?;
                self.variables.insert(name.name.clone(), alloca);
                Ok(())
            }
            Stmt::Assign { target, value, .. } => {
                let val = self.compile_expr(value)?;
                let ptr = self.compile_lvalue(target)?;
                self.builder.build_store(ptr, val)?;
                Ok(())
            }
            Stmt::If { cond, then_block, else_block, .. } => {
                let cond_val = self.compile_expr(cond)?.into_int_value();
                
                let then_bb = self.context.append_basic_block(self.current_function.unwrap(), "then");
                let else_bb = if else_block.is_some() {
                    Some(self.context.append_basic_block(self.current_function.unwrap(), "else"))
                } else {
                    None
                };
                let merge_bb = self.context.append_basic_block(self.current_function.unwrap(), "merge");

                if let Some(else_bb) = else_bb {
                    self.builder.build_conditional_branch(cond_val, then_bb, else_bb)?;
                } else {
                    self.builder.build_conditional_branch(cond_val, then_bb, merge_bb)?;
                }

                // then 块
                self.builder.position_at_end(then_bb);
                for s in then_block { self.compile_stmt(s)?; }
                self.builder.build_unconditional_branch(merge_bb)?;

                // else 块
                if let Some(else_bb) = else_bb {
                    self.builder.position_at_end(else_bb);
                    if let Some(block) = else_block {
                        for s in block { self.compile_stmt(s)?; }
                    }
                    self.builder.build_unconditional_branch(merge_bb)?;
                }

                self.builder.position_at_end(merge_bb);
                Ok(())
            }
            Stmt::Return(expr, _) => {
                if let Some(expr) = expr {
                    let val = self.compile_expr(expr)?;
                    self.builder.build_return(Some(&val))?;
                } else {
                    self.builder.build_return(None)?;
                }
                Ok(())
            }
            Stmt::Expr(expr, _) => {
                self.compile_expr(expr)?;
                Ok(())
            }
            // ... 其他语句
            _ => Err(CodeGenError::Unsupported("语句暂不支持".to_string())),
        }
    }

    fn compile_expr(&mut self, expr: &Expr) -> Result<BasicValueEnum<'ctx>, CodeGenError> {
        match expr {
            Expr::IntLit(n, _) => {
                Ok(self.context.i64_type().const_int(*n as u64, true).into())
            }
            Expr::FloatLit(n, _) => {
                Ok(self.context.f64_type().const_float(*n).into())
            }
            Expr::StringLit(s, _) => {
                let global = self.builder.build_global_string_ptr(s, "str")?;
                Ok(global.as_basic_value_enum())
            }
            Expr::BoolLit(b, _) => {
                Ok(self.context.bool_type().const_int(*b as u64, false).into())
            }
            Expr::Ident(ident) => {
                let ptr = self.variables.get(&ident.name)
                    .ok_or_else(|| CodeGenError::LlvmError(format!("未定义变量: {}", ident.name)))?;
                Ok(self.builder.build_load(*ptr, &ident.name)?)
            }
            Expr::BinaryOp { op, left, right, .. } => {
                let l = self.compile_expr(left)?;
                let r = self.compile_expr(right)?;
                match op {
                    BinaryOp::Add => {
                        if l.is_int_value() {
                            Ok(self.builder.build_int_add(l.into_int_value(), r.into_int_value(), "add")?.into())
                        } else if l.is_float_value() {
                            Ok(self.builder.build_float_add(l.into_float_value(), r.into_float_value(), "add")?.into())
                        } else {
                            Err(CodeGenError::Unsupported("加法类型不支持".to_string()))
                        }
                    }
                    BinaryOp::Gt => {
                        if l.is_int_value() {
                            Ok(self.builder.build_int_compare(
                                inkwell::IntPredicate::SGT,
                                l.into_int_value(),
                                r.into_int_value(),
                                "gt"
                            )?.into())
                        } else if l.is_float_value() {
                            Ok(self.builder.build_float_compare(
                                inkwell::FloatPredicate::OGT,
                                l.into_float_value(),
                                r.into_float_value(),
                                "gt"
                            )?.into())
                        } else {
                            Err(CodeGenError::Unsupported("比较类型不支持".to_string()))
                        }
                    }
                    // ... 其他运算符
                    _ => Err(CodeGenError::Unsupported(format!("运算符 {:?} 暂不支持", op))),
                }
            }
            Expr::Call { func, args, .. } => {
                if let Expr::Ident(ident) = &**func {
                    let fn_val = self.module.get_function(&ident.name)
                        .ok_or_else(|| CodeGenError::LlvmError(format!("未定义函数: {}", ident.name)))?;
                    let compiled_args: Vec<BasicValueEnum> = args
                        .iter()
                        .map(|arg| self.compile_expr(arg))
                        .collect::<Result<_, _>>()?;
                    Ok(self.builder.build_call(fn_val, &compiled_args, "call")?
                        .try_as_basic_value()
                        .left()
                        .unwrap_or_else(|| self.context.i64_type().const_int(0, false).into()))
                } else {
                    Err(CodeGenError::Unsupported("仅支持直接函数调用".to_string()))
                }
            }
            // ... 其他表达式
            _ => Err(CodeGenError::Unsupported("表达式暂不支持".to_string())),
        }
    }

    fn compile_lvalue(&mut self, expr: &Expr) -> Result<PointerValue<'ctx>, CodeGenError> {
        match expr {
            Expr::Ident(ident) => {
                self.variables.get(&ident.name)
                    .copied()
                    .ok_or_else(|| CodeGenError::LlvmError(format!("未定义变量: {}", ident.name)))
            }
            Expr::Field { target, field, .. } => {
                let struct_ptr = self.compile_lvalue(target)?;
                // 计算字段偏移并返回指针
                // 实际实现需要结构体布局信息
                Ok(struct_ptr)
            }
            Expr::Index { target, index, .. } => {
                let list_ptr = self.compile_lvalue(target)?;
                let idx = self.compile_expr(index)?.into_int_value();
                // 计算元素地址
                Ok(unsafe { self.builder.build_gep(list_ptr, &[idx], "index")? })
            }
            _ => Err(CodeGenError::Unsupported("非法的左值".to_string())),
        }
    }
}
```

### 11.5.1 GPU 后端接口

```rust
pub enum GpuTarget {

pub struct GpuBackend {
    target: GpuTarget,
    options: CodeGenOptions,
}

impl CodeGenerator for GpuBackend {
    fn new(target_triple: TargetTriple, options: CodeGenOptions) -> Self {
        Self {
            target: GpuTarget::from_triple(&target_triple),
            options,
```

## 11.6 链接器集成

```rust
pub fn link_objects(objects: &[PathBuf], output: &Path, triple: &str) -> Result<(), LinkError> {
    // 根据目标选择链接器
    let linker = if triple.contains("windows") {
        "link.exe"
    } else if triple.contains("wasm") {
        "wasm-ld"
    } else {
        "ld"  // 或 lld
    };

    let mut cmd = std::process::Command::new(linker);
    for obj in objects {
        cmd.arg(obj);
    }
    cmd.arg("-o").arg(output);

    // 添加运行时库
    if !triple.contains("none") {
        cmd.arg("-lhuanrt");
    }

    let status = cmd.status().map_err(|e| LinkError::LinkFailed(e.to_string()))?;
    if status.success() {
        Ok(())
    } else {
        Err(LinkError::LinkFailed(format!("链接器退出码: {}", status.code().unwrap_or(-1))))
    }
}
```

## 11.7 JIT 执行引擎

幻语支持通过 LLVM JIT 直接执行代码，用于 REPL 和快速测试。

```rust
pub struct JitEngine<'ctx> {
    execution_engine: ExecutionEngine<'ctx>,
}

impl<'ctx> JitEngine<'ctx> {
    pub fn new(module: Module<'ctx>) -> Result<Self, CodeGenError> {
        let execution_engine = module
            .create_jit_execution_engine(inkwell::OptimizationLevel::None)
            .map_err(|e| CodeGenError::LlvmError(e.to_string()))?;
        Ok(Self { execution_engine })
    }

    pub fn run_function<T>(&self, name: &str, args: &[T]) -> Result<T, CodeGenError> {
        unsafe {
            let func: JitFunction<unsafe extern "C" fn(...) -> T> = self.execution_engine
                .get_function(name)
                .map_err(|e| CodeGenError::LlvmError(e.to_string()))?;
            Ok(func.call(args))
        }
    }
}
```
